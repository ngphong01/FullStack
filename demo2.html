:root - Chọn phần tử gốc (). :root { background-color: red; padding: 30px; } :empty - Chọn phần tử không có nội dung. div:empty { border: 1px solid blue; } :checked - Chọn checkbox/radio đang được chọn. input[type="checkbox"]:checked { outline: 2px solid green; } :disabled - Chọn phần tử bị vô hiệu hóa. input:disabled { background-color: #f1f1f1; } :enabled - Chọn phần tử không bị vô hiệu hóa. input:enabled { border: 1px solid dodgerblue; } :focus - Chọn phần tử đang được focus. input:focus { border-color: orange; } :hover - Chọn phần tử khi chuột di qua. a:hover { color: red; } :visited - Chọn liên kết đã truy cập. a:visited { color: purple; } :target - Chọn phần tử được liên kết qua URL fragment. #section:target { background-color: yellow; } :lang() - Chọn phần tử theo ngôn ngữ. p:lang(en) { color: blue; }
